rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAuthenticated() {
      return request.auth != null;
    }

    function isAdmin() {
      return isAuthenticated() &&
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    function isTechnicianOrAdmin() {
      return isAuthenticated() &&
             (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin' ||
              get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'technicien');
    }

    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }

    match /users/{uid} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && request.auth.uid == uid;
      allow update: if isOwner(uid) || isAdmin();
      allow delete: if isAdmin();
    }

    match /TP/{tpId} {
      allow read: if true;
      allow create: if isAuthenticated();
      allow update: if isOwner(resource.data.createdBy) || isAdmin();
      allow delete: if isOwner(resource.data.createdBy) || isAdmin();
    }

    match /machines/{machineId} {
      allow read: if true;
      allow create, update, delete: if isTechnicianOrAdmin();
    }

    match /stocks/{stockId} {
      allow read: if true;
      allow create, update, delete: if isTechnicianOrAdmin();
    }

    match /maintenance/{maintenanceId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update, delete: if isOwner(resource.data.createdBy) || isTechnicianOrAdmin();
    }

    match /activities/{activityId} {
      allow read: if isAdmin();
      allow create: if isAuthenticated();
      allow update, delete: if isAdmin();
    }

    match /consignes/{id} {
      allow read: if isAuthenticated();
      allow create, update, delete: if isAdmin();
    }

    match /zones/{id} {
      allow read: if isAuthenticated();
      allow create, update, delete: if isAdmin();
    }

    match /alertes/{id} {
      allow read: if isAuthenticated();
      allow create, update, delete: if isAdmin();
    }

    match /emplacements/{id} {
      allow read: if isAuthenticated();
      allow create, update, delete: if isTechnicianOrAdmin();
    }

    match /categories/{id} {
      allow read: if isAuthenticated();
      allow create, update, delete: if isTechnicianOrAdmin();
    }

    match /rapports/{id} {
      allow read: if isAuthenticated();
      allow create, update, delete: if isAdmin();
    }

    match /documents/{id} {
      allow read: if isAuthenticated();
      allow create, update, delete: if isAdmin();
    }

    match /settings/{settingId} {
      allow read: if isAuthenticated();
      allow write: if isAdmin();
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
